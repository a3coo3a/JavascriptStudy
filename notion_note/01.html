구문 및 주석

구문

문장 끝에 세미콜론(;)을 입력한다.

구문사이를 구분하기 위해 사용.

생략해도 무관하나, 오류 방지를 위해 사용을 권장한다.

주석

// : 한 줄 주석 처리. 프로그램 코드 끝에 사용

/*내용*/ : 한 줄 이상의 주석 처리

변수

변수 선언

var 키워드

자바스크립트에서 변수의 스코프(scope)는 코드의 블록({})이 아닌 함수 단위

var num = 10;
  var num = 20;
  // 에러 없음. 
  console.log(num);     // 결과 값 : 20

let 키워드

ES6에 추가된 변수

var는 동일 변수의 중복을 허용하지만, let은 중복 변수의 선언을 허용하지 않음

let num = 10;
  let num = 20;
  // 중복 변수 선언을 허용하지 않으므로 에러 발생

데이터 타입

기본타입

숫자는 개발자모드(F12) 에서 파란색으로 표기된다.

var num1 = "10"; //문자
  var num2 = '10'; //문자
  var num3 = 10; //숫자
  
  num1 = "20";/* 새로만든다 생각하면됨. */
  num2 = '20';
  num3 = "30";
  
  console.log(num1);   //20
  console.log(num2);   //20
  console.log(num3);   //30
  
  console.log(num1 + num2);    //2020
  console.log(40 + num1);     //4020
  
  var result = num1 + num2 + num3 + 100;    
  console.log(result);     //202030100

숫자 (number)

정수, 실수 구분 없이 사용

var i1 = 10;
  var i2 = 20;
  var i3 = i1 + i2;
  console.log(i3);    //30

문자열 (string)

문자, 문자열 구분 없이 사용

" " or ' ' 로 묶어서 사용

var s1 = "나는 ";
  var s2 = '너에게 ';
  var s3 = "'안녕!' 이라고 말했다";
  console.log(s1 + s2 + s3);      
  // 결과 : 나는 너에게 '안녕!' 이라고 말했다

불린 (boolean)

false : 0, null, ""(빈문자열), false, undefined, NaN

true : false 외의 값

var boolean1 = true;
  var boolean2 = false;
  console.log(boolean1);   //true
  console.log(boolean2);   //false

undefined

변수는 선언 하였으나 초기화를 하지 않은 경우 자동 저장

반환값

  • boolean : false
  • number : NaN
  • string : "undefined"
var nu;
  console.log(nu);    //undefined

null

값은 저장했으나 존재하지 않을 때 나타냄

var tag = document.querySelector("li");
  console.log(tag);   //null

연산자

기본적으로 자바와 동일 하게 사용됨.

단, 비교 연산자에 다른 부분이 있음.

산술연산자

+, -, *, %, ++, --

비교연산자

동일한 부분

<, >, <=, >=

다른 부분

==, != : 값의 비교

===, !== : 값 + 데이터 타입 비교

  • JS에서 비교 연산자를 쓸때 3개짜리를 쓰는것을 좀 더 권고하고 있다.
var num1 = "10";
  var num2 = 10;
  
  console.log(num1 == num2);  /* true - 값만 비교 */
  console.log(num1 === num2); /* false - 값과 데이터 타입까지 동일한지 비교 */
  
  console.log(num1 != num2); /* false - 값 그 자체를 비교하기 때문에 */
  console.log(num1 !== num2); /* true - 값은 같지만 데이터 타입이 다르기 때문에  */
var a;  /* undefined */
  var b = null;  /* null */
  
  console.log(a, b, a == b, a === b, typeof a, typeof b);  
  /* 둘다 없는 값이므로 == 는 true
     둘의 데이터 타입은 undefined, object로 다르기 때문에 false */
  
  console.log(a != b, a !== b);
  /* false - 없는 것으로 값이 같기에
     true - 값은 같지만 데이터 타입이 다른것이 맞기에 */

대입연산자

=, +=, -=, *=, %=, /=

논리연산자

!, &&, ||, &, |

삼항연산자

조건식 ? 실행문1 : 실행문2

배열

일반적으로 자바의 리스트와 비슷한 형태 - 크기가 자유롭게 늘고 줄어듬

배열의 선언, 생성

기본형태

var arr = [];       // 빈배열
  var arr = [1,2,3];  // 배열형태
  var arr2 = ["안녕","헬로","니하오"];     // 문자형태

서로 다른 타입 선언

var arr3 = [1,2,3,4,5,"hello"];    //가능

객체 생성 선언

var arr1 = new Array();    // var arr = []; 와 같은 의미
  
  // 크기가 10인 빈공간의 배열
  var arr2 = new Array(10);
  
  // 생성과동시에 초기화
  var arr3 = new Array('a','b','c','d','e');

배열의 사용

호출

var arr = [1,2,3];
  console.log(arr) 
arr[0] = 10;
  arr[1] = 20;
  console.log(arr[0], arr[1], arr[2]);

쭉이어서 메소드 확인 후 마지막 결과 표기

length() - 배열의 길이

var arr = new Array('a','b','c','d','e');
  console.log('길이:' + arr.length);    // 길이 : 5

push() - 마지막에 값 추가

arr.push(100);
  arr.push(200);
  console.log(arr);

pop() - 마지막에 값 제거

arr.pop();
  console.log(arr);

unshift() - 앞에서 값 추가

arr.unshift(100);
  console.log(arr);

shift() - 앞에서 값 제거

arr.shift();
  console.log(arr);

splice() - 배열의 중간에 추가, 삭제

// splice(index, 삭제 갯수)
  // 1번 인덱스에서 1개삭제
  arr.splice(1,1);      
  console.log(arr);
  
  // 1번 인덱스에서 2개 삭제
  arr.splice(1, 2);      
  console.log(arr);
// splice(index, 삭제갯수, +추가할 값)
  arr.splice(1, 0, 'park', 'kim', 'lee');
  console.log(arr);

sort() - 정렬

arr.sort();
  console.log(arr);

reverse() - 역정렬

arr.reverse();
  console.log(arr);

indexof() - 탐색

var find = arr.indexOf("a");
  console.log("a가 존재하는 위치 :", find);
  // 결과 ) a가 존재하는 위치 : 4

반복문

자바와 사용방법이 동일 하여 추가된 for in 부분만을 살펴보았다.

while / do while / for

for in

기본 배열에서의 사용법

var arr = [1,2,3,4,5, "홍길동", "이순신"];
      
      // 주의 : 배열의 객체의 index or key값을 변수에 저장
      for (var i in arr) {
        console.log(i);   // index가 출력됨.
      }
  
      for(var i in arr){
        console.log(arr[i]);   // 값
      }

객체형태에서의 사용법

// 자바스크립드의 객체형태에서는 키를 뽑는다.
      var obj = {"key1":"가","key2":"나","key3":"다",}
      console.log(obj);
  
      for(var i in obj){
        console.log(i);  /* key값 */
      }
      for(var i in obj){
        console.log(obj[i]);  /* key에 해당하는 값 */
      }
